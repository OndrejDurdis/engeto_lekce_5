Toto je upráva provedená ve REMOTE na GitHubu.


# Naklonovani remote depozitare (Provadime 1x)
git clone http://url.naseho.repozitare

# Stahnuti aktualniho stavu ze serveru (z GitHubu)
git pull

 # Pridani souboru do STAGE stavu (hromadku souboru pripravenych ke commitu)
git add .

# Vytvoreni commit se zpravou
git commit -m "zprava"

# Odeslani commitu zpatky na server (REMOTE)
git push
